\section{Construction par sous-ensemble}

\setframetitle{Construction par sous-ensemble}

\begin{frame}{\myframetitle}
  \begin{example}[\(M \in \NFA(\{a, b\}, \mathbb{N})\)]
    \centering
    \begin{tikzpicture}
      \node[state, initial] (0) {\(0\)};
      \path let \p0=($(0.north east)-(0.south west)$) in
      node[state, right of=0] (1) {\(1\)}
      node[state, right of=1, above={-0.4 * \y0} of 1, accepting] (2) {\(2\)}
      node[state, right of=1, below={-0.4 * \y0} of 1] (3) {\(3\)};

      \draw
      (0) edge[loop above] node{\(a\)} (0)
      (0) edge[below] node{\(a\)} (1)
      (1) edge[above] node{\(b\)} (2)
      (1) edge[below] node{\(b\)} (3);
    \end{tikzpicture}
  \end{example}
\end{frame}
